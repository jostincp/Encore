app.get('/api/queue/:barId/now-playing', async (req, res) => {
  try {
    const { barId } = req.params;
    
    log('üéµ Getting now playing song', { barId });
    
    const nowPlayingKey = queue::nowPlaying;
    const rawSong = await redis.get(nowPlayingKey);
    
    if (!rawSong) {
      return res.json({
        success: true,
        data: null,
        message: 'No song currently playing'
      });
    }
    
    const song = JSON.parse(rawSong);
    
    log('‚úÖ Now playing song retrieved', { songId: song.id, title: song.title });
    
    return res.json({
      success: true,
      data: song
    });
    
  } catch (error) {
    log('‚ùå Error getting now playing song:', error);
    
    return res.status(500).json({
      success: false,
      message: 'Failed to get now playing song',
      error: error.message
    });
  }
});
